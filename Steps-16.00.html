Creating All Measures Table 16

1.  In Home Tab, select Enter data and Create New table
2.  Rename the table to ‘All Measures’ Table
3.  Click Load
4.  Right Click on 'All Measures' and select New Measure
5.  Copy and paste the DAX formula for Total Revenue  Measure on the Dax Bar 
6.  Click on the Right mark
7.  Repeat step 4-6 until you create all measures using the below Dax formula.
Follow the following Gif for step by step instruction.


1.Total Revenue =

 SUMX('US_Regional_Sales', 'US_Regional_Sales'[Order Quantity] * 'US_Regional_Sales'[Unit Price])


2.RevenueYoY% =

DIVIDE(

    [Total Revenue] -

    CALCULATE([Total Revenue], DATEADD('Calendar Year'[Date], -1, YEAR)),

    CALCULATE([Total Revenue], DATEADD('Calendar Year'[Date], -1, YEAR))

)

 

3.RevenueYoY%_Arrow =

SWITCH(

    TRUE(),

    [RevenueYoY%] < 0, UNICHAR(9660),  // Down arrow for negative change

    [RevenueYoY%] >= 0, UNICHAR(9650)  // Up arrow for positive or zero change

)


4.Total Discount Applied =

SUMX(

    'US_Regional_Sales',

    'US_Regional_Sales'[Order Quantity] *

    'US_Regional_Sales'[Unit Price] *

    'US_Regional_Sales'[Discount Applied]

)



5.Discount YoY% =

DIVIDE(

    [Total Discount Applied] -

    CALCULATE([Total Discount Applied], DATEADD('Calendar Year'[Date], -1, YEAR)),

    CALCULATE([Total Discount Applied], DATEADD('Calendar Year'[Date], -1, YEAR))

)

 

6.Discount YoY%_Arrow =

SWITCH(

    TRUE(),

    [Discount YoY%] < 0, UNICHAR(9660),  // Down arrow for a decrease

    [Discount YoY%] >= 0, UNICHAR(9650)  // Up arrow for an increase or no change

)

 

7.Average Unit Price = AVERAGE('US_Regional_Sales'[Unit Price])

8.AverageUnitPriceYoY% =

DIVIDE(

    [Average Unit Price] -

    CALCULATE([Average Unit Price], DATEADD('Calendar Year'[Date], -1, YEAR)),

    CALCULATE([Average Unit Price], DATEADD('Calendar Year'[Date], -1, YEAR))

)

9.AvgUnitPriceYoY%_Arrow =

SWITCH(

    TRUE(),

    [AvgUnitPriceYoY%] < 0, UNICHAR(9660),  // Down arrow for negative change

    [AvgUnitPriceYoY%] >= 0, UNICHAR(9650)  // Up arrow for positive or zero change

)

 

 

10.Average Order Quantity = AVERAGE('US_Regional_Sales'[Order Quantity])

11.AvgOrderQtyYoY% =

DIVIDE(

    [Average Order Quantity] -

    CALCULATE([Average Order Quantity], DATEADD('Calendar Year'[Date], -1, YEAR)),

    CALCULATE([Average Order Quantity], DATEADD('Calendar Year'[Date], -1, YEAR))

)

12.AvgOrderQtyYoY%_Arrow =

SWITCH(

    TRUE(),

    [AvgOrderQtyYoY%] < 0, UNICHAR(9660),  // Down arrow for negative change

    [AvgOrderQtyYoY%] >= 0, UNICHAR(9650)  // Up arrow for positive or zero change

)

 

13.Total Orders by Channel = COUNTROWS('US_Regional_Sales')

14.Total Sales by Product =

SUMX('US_Regional_Sales', 'US_Regional_Sales'[Order Quantity] * 'US_Regional_Sales'[Unit Price])

 

 

15.ROI-InStore =

VAR InStoreRevenue = 33500000  

VAR RevenueIncrease = InStoreRevenue * 0.05  -- 5% increase in revenue

VAR ROI = RevenueIncrease / InStoreRevenue  -- Calculate ROI

 

RETURN

IF(

    HASONEFILTER('US_Regional_Sales'[Sales Channel]),

    IF(

        SELECTEDVALUE('US_Regional_Sales'[Sales Channel]) = "In-Store",

        ROI,

        BLANK()

    ),

    ROI  -- Display ROI when no filter is applied (unfiltered dashboard)

)

 

 

16.ROI-Online =

VAR OnlineRevenue = 24749599  

VAR RevenueIncrease = OnlineRevenue * 0.07  -- 7% increase in revenue

VAR ROI = RevenueIncrease / OnlineRevenue  -- Calculate ROI

 

RETURN

IF(

    HASONEFILTER('US_Regional_Sales'[Sales Channel]),

    IF(

        SELECTEDVALUE('US_Regional_Sales'[Sales Channel]) = "Online",

        ROI,

        BLANK()

    ),

    ROI  -- Display ROI when no filter is applied (unfiltered dashboard)

)
